`magnitudeToGlowScale` in `coordinates.ts` (same pattern as existing `magnitudeToSize`):

```typescript
export function magnitudeToGlowScale(magnitude: number): number {
  const minMag = -1.5;
  const maxMag = 6.0;
  const maxGlow = 4.0;
  const minGlow = 1.5;
  const normalized = Math.max(
    0,
    Math.min(1, (maxMag - magnitude) / (maxMag - minMag)),
  );
  return minGlow + normalized * (maxGlow - minGlow);
}
```

`StarGlow.tsx` — add optional `scale` prop, default 2.5 (backward compatible):

```typescript
interface StarGlowProps {
  color: string;
  size: number;
  intensity?: number;
  scale?: number;
}
export function StarGlow({ color, size, intensity = 0.6, scale = 2.5 }: StarGlowProps) {
  return (
    <mesh scale={scale}>
      <sphereGeometry args={[size, 16, 16]} />
      <meshBasicMaterial color={color} transparent opacity={intensity * 0.3} blending={THREE.AdditiveBlending} depthWrite={false} side={THREE.BackSide} />
    </mesh>
  );
}
```

`Star.tsx` — compute and pass glow scale:


**useStarData.test.ts:**

```typescript
import { renderHook } from "@testing-library/react";
import { useStarData } from "../useStarData";

describe("useStarData", () => {
  it("returns positioned stars", () => {
    const { result } = renderHook(() => useStarData());
    expect(result.current.stars.length).toBeGreaterThan(0);
    expect(result.current.stars[0]).toHaveProperty("x");
    expect(result.current.stars[0]).toHaveProperty("y");
    expect(result.current.stars[0]).toHaveProperty("z");
  });

  it("getStarById returns correct star", () => {
    const { result } = renderHook(() => useStarData());
    const sirius = result.current.getStarById("sirius");
    expect(sirius?.name).toBe("Sirius");
  });

  it("getStarById returns undefined for unknown id", () => {
    const { result } = renderHook(() => useStarData());
    expect(result.current.getStarById("nonexistent")).toBeUndefined();
  });
});
```

**useConstellationData.test.ts:** same pattern — test `constellations.length > 0`, `getConstellationById("orion")` returns a match, unknown id returns undefined.
